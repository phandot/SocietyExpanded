ig_slavers_guild = {
	color = hsv{ 0.83 0.30 0.59 }
	texture = "gfx/interface/icons/ig_icons/slaversunion2.dds"

	ideologies = {
		ideology_reactionary
		ideology_pro_slavery
		ideology_plutocratic
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_slave_stability
		ig_trait_collective_obligation
		ig_trait_harsh_measures
	}
	
	on_enable = {}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		OR = {
			is_pop_type = aristocrats
			is_pop_type = clergymen
			is_pop_type = officers
			is_pop_type = soldiers
			is_pop_type = farmers
			is_pop_type = shopkeepers
			is_pop_type = capitalists
		}		
	}
	
	pop_weight = {
		value = 0
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 50
			}
		}

		add = {
			desc = "POP_CLERGYMEN"
			if = {
				limit = { 
					is_pop_type = clergymen
				}
				value = 20
			}
		}

		add = {
			desc = "POP_OFFICERS"
			if = {
				limit = { 
					is_pop_type = officers
				}
				value = 50
			}
		}	

		add = {
			desc = "POP_FARMERS"
			if = {
				limit = { 
					is_pop_type = farmers
				}
				value = 20
			}
		}

		add = {
			desc = "POP_SHOPKEEPERS"
			if = {
				limit = { 
					is_pop_type = shopkeepers
				}
				value = 20
			}
		}			
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 35
			}
		}
		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
				}
				value = 15
			}
		}
		if = {
			limit = { 
				owner = {
					OR = {
						has_law = law_type:law_debt_slavery
						has_law = law_type:law_slave_trade
					}
				}
			}
			add = {
				value = 35
			}	
		}
		if = {
			limit = { 
				owner = {
					OR = {
						has_law = law_type:law_legacy_slavery
					}
				}
			}
			add = {
				value = 25
			}	
		}
		if = {
			limit = { 
				owner = {
					OR = {
						has_law = law_type:law_slavery_banned
					}
				}
			}
			subtract = {
				value = 20
			}	
		}
		if = {
			limit = { 
				literacy_rate < 0.35
				culture_accepted = yes
				religion_accepted = yes
			}
			add = { 
				value = 30
			}
		}
		if = {
			limit = { 
				literacy_rate < 0.45
				culture_accepted = yes
				religion_accepted = yes
			}
			add = { 
				value = 20
			}
		}
		if = {
			limit = { 
				literacy_rate < 0.55
				culture_accepted = yes
				religion_accepted = yes
			}
			add = { 
				value = 15
			}
		}
		if = {
			limit = { 
				exists = ig:ig_slavers_guild
				ig:ig_slavers_guild = { is_marginal = no }
			}
			add = { 
				value = 15
			}
		}
		if = {
			limit = { 
				exists = ig:ig_slavers_guild
				ig:ig_slavers_guild = { is_powerful = yes }
			}
			add = { 
				value = 15
			}
		}			
		if = {
			limit = { 
				OR = {
					culture_accepted = no
					religion_accepted = no
				}	
			}
			subtract = { 
				value = 100
			}
		}

		if = {
			limit = {
				exists = ruler
				ruler = {
					OR = {
						has_ideology = ideology_jingoist_leader
						has_ideology = ideology_fascist
						has_ideology = ideology_ethno_nationalist
						has_ideology = ideology_slaver
						has_ideology = ideology_traditionalist
					}
				}
			}
			add = { 
				value = 25
			}
		}
		if = {
			limit = { 
				exists = ig:ig_homeland_faction
			}
			multiply = { 
				value = 0.25
			}
		}
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}	
	}
	
	monarch_weight = {
		value = 1.25
	}
	
	commander_weight = {
		value = 1.25
	}	
	
	noble_chance = {
		value = 0.75
	}
	
	female_commander_chance = {	
		value = 0.1
	}
	
	female_politician_chance = {
		value = 0.1

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.05
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.05
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.05
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.35
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
